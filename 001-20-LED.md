# 20 大きな数字を表示しよう

```blocks
function そろばん形式 (値: number, 表示幅: number) {
    if (値 == 0) {
        basic.showLeds(`
            . . # # .
            . # . . #
            . # . . #
            . # . . #
            . . # # .
            `)
        return
    }
    v = 値
    ks = []
    for (let index = 0; index <= 表示幅 - 1; index++) {
        ks.push(v % 10)
        v = Math.trunc(v / 10)
        col = 4 - index
        そろばん１桁(ks[index], col)
    }
}
function そろばん１桁 (値: number, 表示列: number) {
    pattern = []
    pattern.push(base_pattern5[Math.trunc(値 / 5)])
    for (let value of base_pattern[値 % 5]) {
        pattern.push(value)
    }
    for (let index2 = 0; index2 <= 4; index2++) {
        led.plotBrightness(表示列, index2, pattern[index2])
    }
}
let pattern: number[] = []
let col = 0
let ks: number[] = []
let v = 0
let base_pattern5: number[] = []
let base_pattern: number[][] = []
base_pattern = [[0, 0, 0, 0], [0, 0, 0, 160], [0, 0, 160, 160], [0, 160, 160, 160], [160, 160, 160, 160]]
base_pattern5 = [0, 250]
basic.showLeds(`
    . . . . .
    . . . . .
    . . . . .
    . . . . .
    . . . . .
    `)
basic.forever(function () {
    for (let index4 = 0; index4 <= 99999; index4++) {
        そろばん形式(index4, 5)
        basic.pause(50)
    }
})
```

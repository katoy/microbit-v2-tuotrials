# 20 大きな数字を表示しよう


```blocks
function そろばん形式 (値: number, 表示桁: number) {
    v = 値
    ks = []
    for (let index = 0; index <= 表示桁 - 1; index++) {
        k = v % 10
        k5 = Math.trunc(k / 5)
        k1 = k % 5
        v = Math.trunc(v / 10)
        ks.push(k)
        col = 4 - index
        if (ks[index] != prev_ks[index]) {
            for (let index_k = 0; index_k <= 4; index_k++) {
                led.unplot(col, index_k)
            }
        }
        if (k5 >= 1) {
            led.plot(col, 0)
        }
        for (let i2 = 0; i2 <= k1 - 1; i2++) {
            led.plot(col, 4 - i2)
        }
    }
    for (let index_k = 0; index_k <= 表示桁; index_k++) {
        prev_ks[index_k] = ks[index_k]
    }
}
let col = 0
let k1 = 0
let k5 = 0
let k = 0
let ks: number[] = []
let v = 0
let prev_ks: number[] = []
prev_ks = [0, 0, 0, 0, 0]
basic.forever(function () {
    for (let index = 0; index <= 100; index++) {
        そろばん形式(randint(0, 99999), 5)
        basic.pause(500)
    }
})
```

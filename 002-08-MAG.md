function lines (x0: number, y0: number, x1: number, y1: number) {
    delta_x = x1 - x0
    delta_y = y1 - y0
    delta_large = Math.max(Math.abs(delta_x), Math.abs(delta_y))
    dx = delta_x / delta_large
    dy = delta_y / delta_large
    x = x0 + 0.5
    y = y0 + 0.5
    for (let index2 = 0; index2 < delta_large + 1; index2++) {
        disp_star(x, y)
        x += dx
        y += dy
    }
}
